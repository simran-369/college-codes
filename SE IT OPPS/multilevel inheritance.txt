import java.util.*;
class Animal {
    String name;
    public void animalName(){
        System.out.println(name);
    }
}
class Type extends Animal{
    String classification;
    public void animalType(){
    
    System.out.println(classification);
}
}
public class Legs extends Type{
     int legsNum;
    public void legsNumber(){
       
        System.out.println(legsNum);
    }
public static void main(String[]args){
  Legs MyAnimal = new Legs();
  MyAnimal.name="lion";
  MyAnimal.classification="carnivorous";
  MyAnimal.legsNum = 4;
  MyAnimal.animalName();
  MyAnimal.animalType();
  MyAnimal.legsNumber();
}
}
